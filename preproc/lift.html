XTE_LIFT()

Defined in header <xte/preproc/lift.hpp>

Wraps something callable in a lambda object. Automatically includes a noexcept specifier and a requires clause.


Definitions

 #lift 
#define XTE_LIFT(...)                                                   \
	[](auto&&... args) static noexcept(/* ... */) requires(/* ... */) { \
		return __VA_ARGS__(XTE_FWD(args...));                           \
	}

#unary 
#define XTE_LIFT_UNARY(...)                                         \
	[](auto&& arg) static noexcept(/* ... */) requires(/* ... */) { \
		return __VA_ARGS__(XTE_FWD(arg));                           \
	}

#local 
#define XTE_LIFT_LOCAL(...)                                       \
	[&](auto&&... args) noexcept(/* ... */) requires(/* ... */) { \
		return __VA_ARGS__(XTE_FWD(args...));                     \
	}

#infix 
#define XTE_LIFT_INFIX(...)                                                     \
	[](auto&& lhs, auto&& rhs) static noexcept(/* ... */) requires(/* ... */) { \
		return XTE_FWD(lhs) __VA_ARGS__ (XTE_FWD(rhs));                         \
	}

#member 
#define XTE_LIFT_MEMBER(...)                                                        \
	[](auto&& obj, auto&&... args) static noexcept(/* ... */) requires(/* ... */) { \
		return XTE_FWD(obj)__VA_ARGS__(XTE_FWD(args)...);                           \
	}

#var 
#define XTE_LIFT_VAR(...)               \
	[](auto&& obj) static noexcept {    \
		return XTE_FWD(obj)__VA_ARGS__; \
	}



Example

template<typename T>
void templated(T arg) {
	std::println("{}", arg);
}

void call(auto func) {
	func(5);
}

call(XTE_LIFT(templated));
Output: 
5
