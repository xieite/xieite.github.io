xte::visitor {}

Defined in header <xte/func/visitor.hpp>

Derives from multiple functors and inherits their call operators.


Definition

template<typename... Fns>
struct visitor : Fns... {
	using Fns::operator()...;
};

template<typename... Fns>
visitor(Fns...) -> visitor<Fns...>;



Example

auto f = xte::visitor {
	[](int) { std::println("int"); },
	[](double) { std::println("double"); },
	[](bool) { std::println("bool"); }
};

f(3.14159);
Output: 
double
