xieite::term {}

Defined in header <xieite/io/term.hpp>

A class for controlling standard I/O with toggleable "modes" and text styles. Only supports unix-like platforms.


Definition

 
struct term {
	std::FILE* in;
	std::FILE* out;

	term(std::FILE* = stdin, std::FILE* = stdout) noexcept;

	term(const xieite::term&) = delete;

	~term();

	void block(bool) noexcept;

	void echo(bool) noexcept;

	void canon(bool) noexcept;

	void signal(bool) noexcept;

	void proc(bool) noexcept;

	static constexpr std::string fg_code(std::uint8_t, std::uint8_t, std::uint8_t) noexcept;
};


Example

 
int main() {
	xieite::dump(xieite::read(xieite::term("echo Hello, world!", "r").get()));
}
