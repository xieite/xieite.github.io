xieite::t<>

Defined in fragment xieite:t
(header-only: <xieite/meta/t.hpp>)

An shorter alias for std::type_identity {} to make some code slightly neater.


Definition

 
template<typename T>
using t = std::type_identity<T>;


Example

 
import xieite;

template<typename T>
concept foo = xieite::visitor(
	[]<typename U>(xieite::t<U>) { return false; },
	[]<typename Ret, typename... Args>(xieite::t<Ret(Args...)>) { return true; },
	[](xieite::t<void()>) { return false; }
)(xieite::t<T>());

static_assert(!foo<int>);
static_assert(foo<char(char, char)>);
static_assert(!foo<void(void)>);

int main() {}
