xieite::type_list<Ts...>::has<>

Defined in fragment xieite:type_list
(header-only: <xieite/meta/type_list.hpp>)

Checks if a xieite::type_list contains a particular type.


Definition

 
template<typename T, auto comp = []<typename U, std::same_as<U>> {}>
static constexpr bool has = (... || xieite::is_satisfd<comp, T, Ts>);


Example

 
import std;
import xieite;

int main() {
	using list = xieite::type_list<int, char, float, void*>;

	xieite::dump(list::has<int>);
	xieite::dump(list::has<void>);
}
Output: 
true
false
