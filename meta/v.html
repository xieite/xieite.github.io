xieite::v<>

Defined in fragment xieite:v
(header-only: <xieite/meta/v.hpp>)

A helper alias for std::integral_constant {}.


Definition

 
template<auto x>
using v = std::integral_constant<decltype(x), x>;


Example

 
import xieite;

template<auto...>
struct seq {};

template<auto n>
constexpr auto make_seq = ([]<auto... i>(this auto self, xieite::v<i>...) {
	if constexpr (sizeof...(i) < n) {
		return self(xieite::v<i>()..., xieite::v<static_cast<decltype(n)>(sizeof...(i))>());
	} else {
		return seq<i...>();
	}
})();

int main() {
	([]<int... i>(seq<i...>) {
		xieite::dump(i...);
	})(make_seq<5>);
}
Output: 
0 1 2 3 4
