xieite::is_nothrow_move_ctor<>

Defined in fragment xieite:is_nothrow_move_ctor
(header-only: <xieite/meta/is_nothrow_move_ctor.hpp>)

A wrapper around std::is_nothrow_move_constructible {} to specify that a type is move-constructible without throwing.


Declaration

 
template<typename T>
concept is_nothrow_move_ctor = std::is_nothrow_move_constructible_v<T>;


Example

 
import xieite;

struct A {
	A(A&&) noexcept {}
};

struct B {
	B(B&&) {}
};

int main() {
	xieite::dump(xieite::is_nothrow_move_ctor<A>);
	xieite::dump(xieite::is_nothrow_move_ctor<B>);
}
Output: 
true
false
