xieite::is_satisfied_any<>

Defined in header <xieite/trait/is_satisfied_any.hpp>

Specifies that a functor is satisfied by at least one of several template arguments.


Definition

 
template<auto fn, typename... Ts>
concept is_satisfied_any = (... || xieite::is_satisfd<fn, Ts>);


Example

 
int main() {
	static constexpr auto fn = []<std::integral> {};

	xieite::dump(xieite::is_satisfied_any<fn, int, float, short>);
	xieite::dump(xieite::is_satisfied_any<fn, void*, double, std::nullptr_t>);
}
Output: 
true
false
