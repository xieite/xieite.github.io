xieite::rm_rref<>

Defined in header <xieite/trait/rm_rref.hpp>

Removes the rvalue reference from a type.


Declaration

 
template<typename T>
using rm_rref = std::conditional_t<xieite::is_rref<T>, xieite::rm_ref<T>, T>;


Example

 
static_assert(std::same_as<xieite::rm_rref<int>, int>);
static_assert(std::same_as<xieite::rm_rref<int&>, int&>);
static_assert(std::same_as<xieite::rm_rref<int&&>, int>);
